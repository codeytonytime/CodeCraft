function test_getLoc()

	local TonyTurtle = require("TonyTurtle")

	me = TonyTurtle:new()

	local success, result = pcall(function() return me:getLoc() end)
	print(success, result)
	if success then
		print("Current location:", result)
	else
		print("error:", result)
		return false
	end
	return true
end


function test_getDirection()
	
	local TonyTurtle = require("TonyTurtle")

end
