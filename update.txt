local function generateCacheBustedURL(baseURL)
    local timestamp = os.time()
    return baseURL .. "?cache_bust=" .. timestamp
end


-- Enhanced Auto-Update Script
local function downloadFile(url, filename)
	timestamped_url = generateCacheBustedURL(url)
    print("Downloading:", url)
    local response = http.get(timestamped_url)
    if not response then
        print("Error: Failed to connect to URL:", url)
        return false
    end

    local content = response.readAll()
    response.close()

    local file = fs.open(filename, "w")
    if not file then
        print("Error: Unable to write to file:", filename)
        return false
    end

    file.write(content)
    file.close()
    print("Successfully updated:", filename)
    return true
end

local function autoUpdate(files)
    for filename, url in pairs(files) do
        print("Checking file:", filename)
        local success = downloadFile(url, filename)
        if not success then
            print("Update failed for:", filename)
        end
    end
end


-- Define files to update
local files = {
    ["update.lua"] = "https://raw.githubusercontent.com/codeytonytime/CodeCraft/refs/heads/TonyTurtle/update.txt?version=" .. os.time(),
    ["test.lua"] = "https://raw.githubusercontent.com/codeytonytime/CodeCraft/refs/heads/TonyTurtle/test.txt?version=" .. os.time(),

}





-- Start update process
print("Starting update...")
