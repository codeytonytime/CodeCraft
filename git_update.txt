-- Auto-Update Script
function downloadFile(url, filename)
    print("Downloading file from:", url)
    local response = http.get(url)
    if response then
        local content = response.readAll()
        response.close()
        local file = fs.open(filename, "w")
        file.write(content)
        file.close()
        print("File updated:", filename)
    else
        print("Error: Failed to download file from:", url)
    end
end

function autoUpdate(files)
    for filename, url in pairs(files) do
        print("Updating:", filename)
        downloadFile(url, filename)
    end
    print("All files updated!")
end

-- Define your files and their GitHub raw URLs here
local files = {
    ["test_git.lua"] = "https://raw.githubusercontent.com/username/repository/branch/script1.lua",
}

-- Run the update process
autoUpdate(files)

print("Update complete. Restarting main script...")
shell.run("script1.lua") -- Change this to your main script
